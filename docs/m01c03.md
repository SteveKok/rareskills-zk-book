## Practice Problem

```py
import galois
p = 21888242871839275222246405745257275088548364400416034343698204186575808495617

GF = galois.GF(p)

# Question 1
# x * y * z === 0
# Any of x, y, z is zero will satisfy above equation
# x + y + z === 0
# If I can make x + y = p, I can satisfy the above equation too
x1 = GF(1)
y1 = GF(p - 1)
z1 = GF(0)
assert x1 * y1 * z1 == GF(0)
assert x1 + y1 + z1 == GF(0)
print(f"Question 1: x={x1}, y={y1}, z={z1}")

# Question 2
# x ^ 2 + 2x + 3 === 11
# x ^ 2 + 2x - 8 === 0
# (x + 4)(x - 2) === 0
# 2 is one of the solution, -4 should be another one
x2 = GF(p - 4)
assert x2**2 + 2 * x2 + GF(3) == GF(11)
print(f"Question 2: x={x2}")
```
